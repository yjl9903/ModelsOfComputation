\end{problem}

\begin{solution}

    \centering
    \begin{tabularx}{\textwidth}{@{}Y|Y|Y@{}} \hhline
          & 0   & 1   \\ \hline

        3 & 1L4 & 1R3 \\ \hline
        4 & 0L5 & 1L4 \\ \hline
        5 & 1R1 & 1L5 \\ \hhline
    \end{tabularx}
    \caption{解答 5.2}

\end{solution}
